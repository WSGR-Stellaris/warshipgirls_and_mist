	trigger = {
		owner = {
			OR = { 	
				is_country_type = wg_crisis_country
				is_country_type = wg_crisis_sub_gate_country
				is_wg_fallen_empire = yes
				AND = {
					has_authority = auth_warshipgirls
					has_technology = tech_galokiller
				}
			}
		}
	}	
	icon_frame = 3
	default = yes
	
	stop_when_armies_dead = no
	abduct_pops = no

	planet_damage = 5.0
	army_damage = 3.0	
	
	kill_pop_chance = {
		base = 1.50
	}
	min_pops_to_kill_pop = 0
	
	# root = fleet
	# from = planet
	ai_weight = {
		weight = 1
	}
}

wg_doom = {
	trigger = {
		owner = {
			OR = { 	
				is_country_type = wg_crisis_country
				is_country_type = wg_crisis_sub_gate_country
				has_authority = auth_warshipgirls
				has_authority = auth_shenhai
			}
			AND = {
				has_technology = tech_wg_doom
			}			
		}
	}
	
	icon_frame = 3
	default = no
	
	stop_when_armies_dead = no
	abduct_pops = no

	planet_damage = 8.0
	army_damage = 0.5	
	
	kill_pop_chance = {
		base = 0.5
	}
	min_pops_to_kill_pop = 0
	
	# root = fleet
	# from = planet
	ai_weight = {
		weight = 10
		modifier = {
			factor = 0.01
			exists = from
			from = { 
				owner = { NOT = { is_hostile = root.owner } }
			}
		}
	}
}
