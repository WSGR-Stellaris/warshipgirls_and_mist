building_sh_research = {
	base_buildtime = 300

	category = research
	
	potential = {
		exists = owner
		OR = {
			owner = { is_ai = no }
			free_building_slots > 0
		}
		owner = { 
			is_regular_empire = yes
			has_authority = auth_shenhai	
			NOT = { has_policy_flag = robots_outlawed }
		}
		NOT = { has_modifier = slave_colony }
	}
	
	allow = {
		has_upgraded_capital = yes
	}
	
	destroy_trigger = {
		free_building_slots = 0
		exists = owner
		owner = {
			is_ai = yes
			has_monthly_income = {
				resource = sr_pantsu
				value > 11
			}
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }					
		}
		modifier = {
			job_researcher_add = 4
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }					
		}
		modifier = {
			job_researcher_add = 4
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 400
		}	
		upkeep = {
			energy = 5
		}
	}	
	upgrades = {
		"building_sh_research2"
	}	
	prerequisites = {
		"tech_sh_research"
	}
	
	triggered_desc = {
		text = job_researcher_effect_desc
	}
	
	ai_weight = {
		weight = 0
		modifier = {
			weight = 50
			owner = { 
				has_monthly_income = {
					resource = consumer_goods
					value > 0
				}
			}
			buildings_build_building = yes
		}
		modifier = {
			weight = 600
			owner = {
				has_monthly_income = {
					resource = consumer_goods
					value > 50
				}
			}
		}
	}
}
