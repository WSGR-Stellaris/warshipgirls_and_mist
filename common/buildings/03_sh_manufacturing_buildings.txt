building_sh_pantsu = {
	base_buildtime = 500

	category = manufacturing
	
	potential = {
		exists = owner
		OR = {
			owner = { is_ai = no }
			free_building_slots > 0
		}
		owner = { 
			is_regular_empire = yes
			has_authority = auth_shenhai	
			NOT = { has_policy_flag = robots_outlawed }
		}
		NOT = { has_modifier = slave_colony }
	}
	
	allow = {
		has_upgraded_capital = yes
	}
	
	destroy_trigger = {
		free_building_slots = 0
		exists = owner
		owner = {
			is_ai = yes
			has_monthly_income = {
				resource = sr_pantsu
				value > 11
			}
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }					
		}
		modifier = {
			job_sh_pantsu_maker_add = 4
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }					
		}
		modifier = {
			job_sh_pantsu_maker_add = 3
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 400
		}	
		upkeep = {
			energy = 4
		}
	}	
	
	prerequisites = {
		"tech_sh_assembly"
	}
	
	triggered_desc = {
		text = sh_pantsu_effect_desc
	}
	
	ai_weight = {
		weight = 0
		modifier = {
			weight = 50
			owner = { 
				has_monthly_income = {
					resource = sr_pantsu
					value < 5
				}
			}
			buildings_build_building = yes
		}
		modifier = {
			weight = 600
			owner = {
				has_monthly_income = {
					resource = sr_pantsu
					value < 2
				}#5 is too high since that means 2 extra buildings. Better to just go with 1 extra building
			}
		}
	}

	ai_resource_production = {
		sr_pantsu = 1
		trigger = {
			always = yes
		}
	}
}
