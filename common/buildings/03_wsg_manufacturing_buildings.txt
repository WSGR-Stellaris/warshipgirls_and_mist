		}

			is_regular_empire = yes

		}

		has_upgraded_capital = yes
	}

		free_building_slots = 0
		exists = owner
		owner = {
			is_ai = yes

				resource = sr_pantsu
				value > 11
			}
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }					
		}
		modifier = {
			job_wsg_pantsu_maker_add = 4
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }					
		}
		modifier = {
			job_wsg_pantsu_maker_add = 3
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 500
		}	
		upkeep = {
			energy = 5
			exotic_gases = 1
		}
	}	
	
	prerequisites = {
		"tech_wsg_pantsu"
	}
	
	triggered_desc = {
		text = pantsu_effect_desc
	}
	
	ai_weight = {
		weight = 0
		modifier = {
			weight = 50
			owner = { 
				has_monthly_income = {
					resource = sr_pantsu
					value < 5
				}
			}
			buildings_build_building = yes
		}
		modifier = {
			weight = 600
			owner = {
				has_monthly_income = {
