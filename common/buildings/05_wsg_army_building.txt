building_wsg_fortress = {
	base_buildtime = 480
	base_cap_amount = 1
	add_to_first_building_slot = yes

	category = army
	
	potential = {
		buildings_simple_allow = yes
	}
	
	destroy_trigger = {
		exists = owner
		owner = { is_ai = yes }
		num_buildings = { type = building_stronghold value > 1 }
	}
	allow = {
		num_pops >= 15
	}	
	planet_modifier = {
		planet_housing_add = 1
		planet_army_build_speed_mult  = 0.20
		planet_orbital_bombardment_damage = -0.50
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			job_soldier_add = 10
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_warrior_drone_add = 10
		}
	}

	resources = {
		category = planet_buildings_strongholds
		cost = {
			alloys = 300
			sr_pantsu = 50
		}			
		upkeep = {
			energy = 10
			sr_pantsu = 1
		}
	}
	
	prerequisites = {
		"tech_building_wsg_fortress"
	}
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_warrior_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_soldier_effect_desc
	}	
	ai_weight = {
		weight = 0
		modifier = {
			weight = 50000
			num_pops >= 40
			NOR = {
				has_building = building_stronghold
				has_building = building_fortress
			}
		}
	}
}
