# 判断大小姐是否能执行项目
country_event = {
	id = wg_lady.7
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = { 
				event_target:lady_contingency_fleet = {
					exists = this
					is_in_combat = no
				}
			}
			random_owned_ship = {
				limit = { fleet = { has_fleet_flag = lady_fleet }}
				set_ship_flag = researching_special_project
				fleet = { save_global_event_target_as = lady_contingency_fleet }
			}
			# set_country_flag = modifing_wsg_lady
			event_target:lady_contingency_fleet = { clear_fleet_actions = this }
			capital_scope = { solar_system = {
				random_system_planet = {
					limit = { NOR = {
						is_same_value = PREVPREV
						is_star = yes
					}}
					create_ambient_object = {
						type = "horror_energy_object"
						location = this
						entity_offset = {
							min = 1
							max = 100
						}
					}
					last_created_ambient_object = {
						save_event_target_as = wg_wormhole
					}
				}
			}}
			enable_special_project = {
				name = wsg_lady_to_wg_fe
				location = event_target:wg_wormhole
			}
			event_target:lady_contingency_fleet = {
				fleet_action_research_special_project = {
					special_project = wsg_lady_to_wg_fe
					target = event_target:wg_wormhole
				}
			}
			event_target:lady_contingency_fleet = { set_event_locked = yes }
			country_event = { id = wg_lady.7001 days = 1 }
			country_event = { id = wg_lady.7002 days = 3 }
		}
		else = {
			owner = {
				abort_special_project = {
					type = pre_wsg_lady_to_wg_fe
				}
				country_event = { id = wg_lady.8 }
			}
		}
	}
}

fleet_event = {
	id = wg_lady.7000
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				OR = {
					NOT = { has_fleet_flag = lady_fleet }
					count_owned_ship = { count > 1 }
					any_owned_ship = {
						NOT = { has_ship_flag = researching_special_project }
					}
				}
			}
			owner = {
				abort_special_project = {
					type = pre_wsg_lady_to_wg_fe
				}
				abort_special_project = {
					type = wsg_lady_to_wg_fe
				}
				country_event = { id = wg_lady.8 }
			}
			event_target:lady_contingency_fleet = {
				set_event_locked = no
				clear_fleet_actions = this
			}
		} else = {
			owner = {
				country_event = { id = wg_lady.7002 }
			}
		}
	}
}

country_event = {
	id = wg_lady.7001
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		NOT = { has_country_flag = finish_wsg_lady_to_wg_fe_project }
		has_special_project = wsg_lady_to_wg_fe
	}

	immediate = {
		event_target:lady_contingency_fleet = { set_event_locked = yes }
		if = {
			limit = {
				has_special_project = wsg_lady_to_wg_fe
				NOT = { is_researching_special_project = wsg_lady_to_wg_fe }
				event_target:lady_contingency_fleet = {
					is_fleet_idle = yes
				}
			}
			if = {
				limit = { exists = event_target:lady_contingency_fleet }
				event_target:lady_contingency_fleet = {
					set_event_locked = no
					clear_fleet_actions = this
	
					fleet_action_research_special_project = {
						special_project = wsg_lady_to_wg_fe
						target = event_target:wg_wormhole
					}
				}
			} 
			country_event = { id = wg_lady.7001 days = 1 }
		} else = {
			country_event = { id = wg_lady.7001 days = 15 }
		}
	}
}
