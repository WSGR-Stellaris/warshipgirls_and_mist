#Increment events - resources
country_event = {
	id = wsgranking.101
	is_triggered_only = yes
	hide_window = yes
	trigger = {
		is_ai = no
		NOT = { has_country_flag = wsgranking101 }
		OR = {
			has_authority = auth_warshipgirls
			has_authority = auth_shenhai
		}
		resource_income_compare = {
			resource = energy
			value >= 100
		}
	}
	immediate = {
		set_country_flag = wsgranking101
		if = {
			limit = { NOT = { has_country_flag = ban_notice_wsgranking }}
			country_event = { id = wsgranking.10101 }
		}
		change_variable = {
			which = rankpts
			value = 2
		}
	}
}
