			add_district = district_farming

			add_district = district_mining
			create_species = {
				name = "NAME_ice_breaking_species_SH"
				class = "ShenHai"
				portrait = "shenhai"
				traits = {
					trait = sh_trait_shenhai3
					trait = sh_trait_together

				effect = { save_global_event_target_as = wsg_ice_breaking_species_sh }

			while = {
				count = 6
				create_pop = {
					species = event_target:wsg_ice_breaking_species_sh
				}	
			}			
		}
	}
}

# 生成鱼唇的sol III
event = {
	id = wsg.4
	hide_window = yes
	is_triggered_only = yes
	fire_only_once = yes

	trigger = {
		any_country = {
			has_authority = auth_warshipgirls
			is_ai = no
		}
	}

	immediate = {
		random_country = {
			limit = {
				has_authority = auth_warshipgirls
				is_ai = no
			}
			capital_scope = {
				closest_system = {
					min_steps = 5
					limit = {
						NOR = {
							exists = space_owner
							has_star_flag = hostile_system
							has_star_flag = guardian
						}
					}
					spawn_system = {
						initializer = "lex_sol_system"
					}
				}
			}
		}
	}

}
